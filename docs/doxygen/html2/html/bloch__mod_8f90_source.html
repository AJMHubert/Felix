417  SUBROUTINE createwavefunctions(RThickness,RFullWaveIntensity,CFullWaveFunctions,&
418  nreflections,nbeams,istrongbeamlist,ceigenvectors,ceigenvalues,ierr)
419 
420  USE mynumbers
421  USE mympi
422  USE message_mod
423 
424  IMPLICIT NONE
425 
426  REAL(RKIND),INTENT(IN) :: RThickness
427  REAL(RKIND),INTENT(OUT) :: RFullWaveIntensity(nreflections)
428  COMPLEX(CKIND),INTENT(OUT) :: CFullWaveFunctions(nreflections)
429  INTEGER(IKIND),INTENT(IN) :: nReflections,nBeams,IStrongBeamList(nreflections)
430  COMPLEX(CKIND),INTENT(IN) :: CEigenVectors(nbeams,nbeams),CEigenValues(nbeams)
431  INTEGER(IKIND),INTENT(OUT) :: IErr
432  REAL(RKIND) :: RWaveIntensity(nbeams)
433  COMPLEX(CKIND) :: CInvertedEigenVectors(nbeams,nbeams),CPsi0(nbeams),&
434  CWaveFunctions(nbeams),CEigenValueDependentTerms(nbeams,nbeams),&
435  CAlphaWeightingCoefficients(nbeams)
436  INTEGER(IKIND) :: ind,jnd,knd,hnd,ifullind,iuniind,gnd,ichnk
437  COMPLEX(CKIND) :: CDummyEigenVectors(nbeams,nbeams)
438 
439  ! The top surface boundary conditions
440  cpsi0 = czero ! all diffracted beams are zero
441  cpsi0(1) = cone ! the 000 beam has unit amplitude
442 
443  ! Invert the EigenVector matrix
444  cdummyeigenvectors = ceigenvectors
445  CALL invert(nbeams,cdummyeigenvectors(:,:),cinvertedeigenvectors,ierr)
446 
447  ! put in the thickness
448  ! From EQ 6.32 in Kirkland Advance Computing in EM
449  calphaweightingcoefficients = matmul(cinvertedeigenvectors(1:nbeams,1:nbeams),cpsi0)
450  ceigenvaluedependentterms= czero
451  DO hnd=1,nbeams ! This is a diagonal matrix
452  ceigenvaluedependentterms(hnd,hnd) = &
453  exp(cimagone*cmplx(rthickness,zero,ckind)*ceigenvalues(hnd))
454  ENDDO
455  ! The diffracted intensity for each beam
456  ! EQ 6.35 in Kirkland Advance Computing in EM
457  ! C-1*C*alpha
458  cwavefunctions(:) = matmul( &
459  matmul(ceigenvectors(1:nbeams,1:nbeams),ceigenvaluedependentterms), &
460  calphaweightingcoefficients(:) )
461 
462  !?? possible small time saving here by only calculating the (tens of) output
463  !?? reflections rather than all strong beams (hundreds)
464  DO hnd=1,nbeams
465  rwaveintensity(hnd)=conjg(cwavefunctions(hnd)) * cwavefunctions(hnd)
466  ENDDO
467 
468  cfullwavefunctions=czero
469  rfullwaveintensity=zero
470  DO knd=1,nbeams
471  cfullwavefunctions(istrongbeamlist(knd))=cwavefunctions(knd)
472  rfullwaveintensity(istrongbeamlist(knd))=rwaveintensity(knd)
473  ENDDO
474 
475  END SUBROUTINE createwavefunctions

516  ! NB RPertStrength0 is an array of perturbation strengths for all reflections

524  rmaxsg = 0.005

526

528  ! now increase RMaxSg until we have enough strong beams
529  DO WHILE (sum(istrong).LT.iminstrongbeams)
530  WHERE (abs(rdevpara).LT.rmaxsg.OR.rpertstrength0.GE.rminpertstrong)
531  istrong=1_ikind
532  END WHERE
533  rmaxsg=rmaxsg+0.005
534  ! RMinPertStrong=0.0025/RMaxSg
535  END DO
536

538  istrongbeamlist=0_ikind
539  ind=1_ikind
540  DO jnd=1,nreflections
541  IF (istrong(jnd).EQ.1) THEN
542  istrongbeamlist(ind)=jnd
543  ind=ind+1
544  END IF
545  END DO

551  CALL message(lxl,dbg7,"Smallest strong perturbation strength = ",rminpertstrong)
552  IF(sum(istrong)+iminweakbeams.GT.nreflections) ierr = 1
553  IF(l_alert(ierr,"StrongAndWeakBeamsDetermination",&
554  "Insufficient reflections to accommodate all Strong and Weak Beams")) RETURN
555 
556  !----------------------------------------------------------------------------
557  ! weak beams
558  !----------------------------------------------------------------------------
559 
560  ! Decrease perturbation strength until we have enough weak beams
561  ! NB IWeak is an array listing the weak beams (1=Weak, 0=Not weak)
562  iweak=0_ikind
563  rminpertweak=0.9*rminpertstrong
564  DO WHILE (sum(iweak).LT.iminweakbeams)
565  WHERE (rpertstrength0.GE.rminpertweak.AND.istrong.NE.1_ikind)
566  iweak=1
567  END WHERE
568  rminpertweak=0.9*rminpertweak
569  END DO
570 
571  CALL message(lxl,dbg7,"weak beams",sum(iweak))
572  CALL message(lxl,dbg7,"Smallest weak perturbation strength = ",rminpertweak)
573 
574  ! give the weak beams a number in IWeakBeamList
575  iweakbeamlist=0_ikind
576  ind=1_ikind
577  DO jnd=1,nreflections
578  IF (iweak(jnd).EQ.1) THEN
579  iweakbeamlist(ind)=jnd
580  ind=ind+1
581  END IF
582  END DO
583  nweakbeams=ind-1
584 
585  CALL message(lxl,dbg7,"Weak Beam List",iweakbeamlist)
