53  SUBROUTINE simulateandfit(RIndependentVariable,Iter,IThicknessIndex,IErr)
54 
55  ! JR (very rough) overview:
56  ! RIndependentVariable holds refinement variables, UpdateVariables updates matching variable
57  ! various global variables which were read from files or setup are now constant
58  ! UniqueAtomPositions used to recalculate all atoms in lattice from basis atoms
59  ! CUgMat = CUgMatNoAbs + CUgMatPrime ( from absoption )
60  ! Simulate ( CUgMat + others ) ---> RImageSimi
61  ! On core 0, FigureOfMeritAndThickness includes image processing
62  ! RImageExpi(x,y,LACBED_ID) are compared to RImageSimi(x,y,LACBED_ID, thickness_ID)
63  ! This calculates ---> RFigureofMerit
64  ! MPI_BCAST(RFigureofMerit) then sends RFigureofMerit to all cores
65 
66  USE mynumbers
67  USE message_mod
68 
69  USE mympi
70  USE ug_matrix_mod
71  USE crystallography_mod
72  USE write_output_mod
73 
74  ! global inputs
75  USE ipara, ONLY : inoofvariables, nreflections, iabsorbflag, inoofugs, &
76  ipixelcount, isimflag, isymmetryrelations, iugoffset, irefinemode, &
77  iequivalentugkey
78  USE rpara, ONLY : rangstromconversion,relectroncharge,relectronmass,&
79  rconvergenceangle, rminimumgmag, rtolerance, rrelativisticcorrection, &
80  rvolume, rgmatrix, rgmatrixmagnitude, rcurrentgmagnitude,rhkl
81  USE rconst, ONLY : rplanckconstant
82 
83  ! global outputs
84  USE cpara, ONLY : cugmat,cugmatnoabs, cuniqueug
85  USE rpara, ONLY : rabsorptionpercentage, rdeltak, rfigureofmerit, rsimulatedpatterns
86 
87  IMPLICIT NONE
88 
89  REAL(RKIND),INTENT(INOUT) :: RIndependentVariable(inoofvariables)
90  INTEGER(IKIND),INTENT(INOUT) :: Iter
91  INTEGER(IKIND),INTENT(OUT) :: IThicknessIndex
92  ! NB IThicknessIndex is calculated and used on rank 0 only
93  INTEGER(IKIND),INTENT(OUT) :: IErr
94  INTEGER(IKIND) :: ind,jnd, ILoc(2), IUniqueUgs
95  INTEGER(IKIND), SAVE :: IStartTime
96  REAL(RKIND) :: RCurrentG(3), RScatteringFactor
97  COMPLEX(CKIND) :: CUgMatDummy(nreflections,nreflections),CVgij
98  CHARACTER*100 :: SFormat,SPrintString
99 
100  WRITE(sprintstring,fmt='(A10,I5)')"Iteration ",iter
101  CALL message(ls,sprintstring)
102  CALL system_clock( istarttime )
103 
104 
105  !\/----------------------------------------------------------------------
106  IF (irefinemode(1).EQ.1) THEN ! Ug refinement; update structure factors
107  ! Dummy Matrix to contain new iterative values
108  cugmatdummy = czero ! NB these are Ug's without absorption
109  jnd=1
110  ! work through the Ug's to update
111  DO ind = 1+iugoffset,inoofugs+iugoffset
112  ! Don't update components smaller than RTolerance:
113  ! 3 possible types of Ug, complex, REAL and imaginary
114  IF ( (abs(REAL(CUniqueUg(ind),RKIND)).GE.rtolerance).AND.&
115  (abs(aimag(cuniqueug(ind))).GE.rtolerance)) then ! use both REAL and imag parts
116  cuniqueug(ind)=cmplx(rindependentvariable(jnd),rindependentvariable(jnd+1))
117  jnd=jnd+2
118  ELSEIF ( abs(aimag(cuniqueug(ind))).LT.rtolerance ) THEN ! use only REAL part
119  cuniqueug(ind)=cmplx(rindependentvariable(jnd),zero)
120  jnd=jnd+1
121  ELSEIF ( abs(REAL(CUniqueUg(ind),RKIND)).LT.rtolerance ) then ! use only imag part
122  cuniqueug(ind)=cmplx(zero,rindependentvariable(jnd))
123  jnd=jnd+1
124  ELSE ! should never happen
125  ierr=1;
126  WRITE(sprintstring,*) ind
127  IF(l_alert(ierr,"SimulateAndFit",&
128  "zero structure factor, CUniqueUg element number="//trim(sprintstring))) RETURN
129  END IF
130  ! Update the Ug matrix for this Ug
131  WHERE(isymmetryrelations.EQ.iequivalentugkey(ind))
132  cugmatdummy = cuniqueug(ind)
133  END WHERE
134  WHERE(isymmetryrelations.EQ.-iequivalentugkey(ind))
135  cugmatdummy = conjg(cuniqueug(ind))
136  END WHERE
137  END DO
138  ! put the changes into CUgMatNoAbs
139  WHERE(abs(cugmatdummy).GT.tiny)
140  cugmatnoabs = cugmatdummy
141  END WHERE
142  CALL absorption(ierr)
143  IF(l_alert(ierr,"SimulateAndFit","Absorption")) RETURN
144  IF (iabsorbflag.EQ.1) THEN ! proportional absorption
145  rabsorptionpercentage = rindependentvariable(jnd)
146  END IF
147  ELSE ! not Ug refinement
148  ! Update variables
149  CALL updatevariables(rindependentvariable,ierr)
150  IF(l_alert(ierr,"SimulateAndFit","UpdateVariables")) RETURN
151  IF (irefinemode(8).EQ.1) THEN ! convergence angle
152  ! recalculate resolution in k space
153  IF (my_rank.EQ.0) rdeltak = rminimumgmag*rconvergenceangle/REAL(ipixelcount,rkind)
154  CALL mpi_bcast(rdeltak,1,mpi_double_precision,0,mpi_comm_world,ierr)
155  ELSE
156  ! basis has changed in some way, recalculate unit cell
157  !PRINT*,"About to UniqueAtomPositions"
158  CALL uniqueatompositions(ierr)
159  IF(l_alert(ierr,"SimulateAndFit","UniqueAtomPositions")) RETURN
160  !--------------------------------------------------------------------
161  ! update scattering matrix Ug
162  !--------------------------------------------------------------------
163  IF (my_rank.EQ.0) then!There is a bug when individual cores calculate UgMat, make it the responsibility of core 0 and broadcast it
164  ! calculate CUgMatNoAbs
165  cugmatnoabs = czero
166  !PRINT*,"About to CUgMatNoAbs"
167  !duplicated from Ug matrix initialisation. Ug refinement will no longer work! Should be put into a single subroutine.
168  DO ind=2,nreflections
169  DO jnd=1,ind-1
170  rcurrentgmagnitude = rgmatrixmagnitude(ind,jnd) ! g-vector magnitude, global variable
171  ! Sums CVgij contribution from each atom and pseudoatom in Volts
172  CALL getvgcontributionij(rscatteringfactor,ind,jnd,cvgij,ierr)
173  cugmatnoabs(ind,jnd)=cvgij
174  ENDDO
175  ENDDO
176  !Convert to Ug
177  cugmatnoabs=cugmatnoabs*two*relectronmass*rrelativisticcorrection*relectroncharge/((rplanckconstant**2)*(rangstromconversion**2))
178  ! NB Only the lower half of the Vg matrix was calculated, this completes the upper half
179  cugmatdummy = transpose(cugmatnoabs)! Dummy just used as a box to avoid the bug when conj(transpose) is used on orac
180  cugmatnoabs = cugmatnoabs + conjg(cugmatdummy)
181  END IF
182  ind=nreflections*nreflections
183  !===================================== ! Send UgMat to all cores
184  CALL mpi_bcast(cugmat,ind,mpi_double_precision,0,mpi_comm_world,ierr)
185  !=====================================
186  CALL absorption(ierr)! calculates CUgMat = CUgMatNoAbs + CUgMatPrime
187  IF(l_alert(ierr,"SimulateAndFit","Absorption")) RETURN
188  END IF
189  END IF
190  !/\----------------------------------------------------------------------
191  CALL message( lm,dbg3, "recalculated Ug matrix, with absorption (nm^-2)" )
192  DO ind = 1,16
193  WRITE(sprintstring,fmt='(3(I2,1X),A2,1X,8(F7.4,1X))') nint(rhkl(ind,:)),": ",100*cugmat(ind,1:4)
194  CALL message( lm,dbg3, sprintstring)
195  END DO
196 
197 
198  IF (my_rank.EQ.0) THEN ! send current values to screen
199  CALL printvariables(ierr)
200  IF(l_alert(ierr,"SimulateAndFit","PrintVariables")) RETURN
201  END IF
202 
203  ! simulate
204  rsimulatedpatterns = zero ! Reset simulation
205  CALL simulate(ierr) ! simulate
206  IF(l_alert(ierr,"SimulateAndFit","Simulate")) RETURN
207 
208  IF(my_rank.EQ.0) THEN
209  ! Only calculate figure of merit if we are refining
210  IF (isimflag.EQ.0) THEN
211  CALL figureofmeritandthickness(iter,ithicknessindex,ierr)
212  IF(l_alert(ierr,"SimulateAndFit","FigureOfMeritAndThickness")) RETURN
213  END IF
214  ! Write current variable list and fit to IterationLog.txt
215  CALL writeoutvariables(iter,ierr)
216  IF(l_alert(ierr,"SimulateAndFit","WriteOutVariables")) RETURN
217  END IF
218 
219  !===================================== ! Send the fit index to all cores
220  CALL mpi_bcast(rfigureofmerit,1,mpi_double_precision,0,mpi_comm_world,ierr)
221  !=====================================
222 
223  END SUBROUTINE simulateandfit

234  SUBROUTINE simulate(IErr)
235 
236  USE mynumbers
237  USE iconst, ONLY : ithree
238  USE mympi
239  USE message_mod
240 
241  USE bloch_mod
242 
243  !global outputs
244  USE rpara, ONLY : rimagesimi, &
245  rsimulatedpatterns
246  ! RImageSimi(x_coordinate, y_coordinate y, LACBED_pattern_ID , thickness_ID )
247  ! RSimulatedPatterns( Pixel_ID, LACBED_pattern_ID , thickness_ID )
248  ! RSimulatedPatterns has a long list of pixel instead of a 2D image matrix
249  USE rpara, ONLY : rindividualreflections
250  USE ipara, ONLY : iinitialsimulationflag, ipixelcomputed
251 
252  !global inputs
253  USE rpara, ONLY : rblurradius
254  USE ipara, ONLY : icount,idisplacements,ilocalpixelcountmax,inooflacbedpatterns,&
255  ilocalpixelcountmin,ipixellocations,ipixelcount,ithicknesscount
256 
257  IMPLICIT NONE
258 
259  INTEGER(IKIND) :: IErr, ind,jnd,knd,pnd, IIterationFLAG, IStartTime
260  INTEGER(IKIND) :: IAbsorbTag = 0
261  REAL(RKIND),DIMENSION(:,:,:),ALLOCATABLE :: RFinalMontageImageRoot
262  REAL(RKIND),DIMENSION(:,:),ALLOCATABLE :: RTempImage
263 
264  ! Reset simuation
265  rindividualreflections = zero
266 
267  CALL system_clock( istarttime )
268 
269  ! Simulation (different local pixels for each core)
270  CALL message(ls,"Bloch wave calculation...")
271  DO knd = ilocalpixelcountmin,ilocalpixelcountmax,1
272  jnd = ipixellocations(knd,1)
273  ind = ipixellocations(knd,2)
274  ! fills array for each pixel number not x & y coordinates
275  CALL blochcoefficientcalculation(ind,jnd,knd,ilocalpixelcountmin,ierr)
276  IF(l_alert(ierr,"Simulate","BlochCoefficientCalculation")) RETURN
277  END DO
278 
279  !===================================== ! MPI gatherv into RSimulatedPatterns
280  CALL mpi_gatherv(rindividualreflections,SIZE(rindividualreflections),mpi_double_precision,&
281  rsimulatedpatterns,icount,idisplacements,mpi_double_precision,&
282  root,mpi_comm_world,ierr)
283  !=====================================
284  IF(l_alert(ierr,"SimulateAndFit","MPI_GATHERV")) RETURN
285 
286  CALL printendtime( lm, istarttime, "Bloch wave simulation" )
287 
288  ! put 1D array RSimulatedPatterns into 2D image RImageSimi
289  ! remember dimensions of RSimulatedPatterns(INoOfLacbedPatterns,IThicknessCount,IPixelTotal)
290  ! and RImageSimi(width, height,INoOfLacbedPatterns,IThicknessCount )
291  rimagesimi = zero
292  ind = 0
293  DO jnd = 1,2*ipixelcount
294  DO knd = 1,2*ipixelcount
295  ind = ind+1
296  rimagesimi(jnd,knd,:,:) = rsimulatedpatterns(:,:,ind)
297  END DO
298  END DO
299 
300  ! Gaussian blur to match experiment using global variable RBlurRadius
301  IF (rblurradius.GT.tiny) THEN
302  ALLOCATE(rtempimage(2*ipixelcount,2*ipixelcount),stat=ierr)
303  DO ind=1,inooflacbedpatterns
304  DO jnd=1,ithicknesscount
305  CALL blurg(rimagesimi(:,:,ind,jnd),ipixelcount,rblurradius,ierr)
306  END DO
307  END DO
308  END IF
309 
310  ! We have done at least one simulation now
311  iinitialsimulationflag=0
312 
313  END SUBROUTINE simulate

324  SUBROUTINE figureofmeritandthickness(Iter,IBestThicknessIndex,IErr)
325 
326  !?? NB this is called on core 0 only
327  USE mynumbers
328  USE message_mod
329 
330  USE utilities_mod, ONLY : phasecorrelate, residualsumofsquares, &
331  normalised2dcrosscorrelation, maskedcorrelation
332 
333  ! global outputs
334  USE rpara, ONLY : rfigureofmerit
335 
336  ! global inputs
337  USE ipara, ONLY : inooflacbedpatterns, icorrelationflag, ipixelcount, ithicknesscount, &
338  iimageprocessingflag
339  USE rpara, ONLY : rinitialthickness, rdeltathickness, rimagemask, &
340  rimagesimi, & ! a main input - simulated images
341  rimageexpi ! a main input - experimental images to compare
342 
343  IMPLICIT NONE
344 
345  INTEGER(IKIND),INTENT(OUT) :: IBestThicknessIndex
346  INTEGER(IKIND) :: ind,jnd,knd,IErr,IThickness,hnd,Iter
347  INTEGER(IKIND),DIMENSION(INoOfLacbedPatterns) :: IBestImageThicknessIndex
348  REAL(RKIND),DIMENSION(2*IPixelCount,2*IPixelCount) :: RSimulatedImage,RExperimentalImage
349  REAL(RKIND),DIMENSION(:,:),ALLOCATABLE :: RMaskImage
350  REAL(RKIND) :: RTotalCorrelation,RBestTotalCorrelation,RImageCorrelation,RBestThickness,&
351  RThicknessRange,Rradius
352  REAL(RKIND),DIMENSION(INoOfLacbedPatterns) :: RBestCorrelation
353  CHARACTER*100 :: SPrintString
354  CHARACTER*20 :: Snum
355 
356  IF (icorrelationflag.EQ.3) THEN ! allocate mask
357  ALLOCATE(rmaskimage(2*ipixelcount,2*ipixelcount),stat=ierr)
358  END IF
359  ! The best correlation for each image will go in here, initialise at the maximum value
360  rbestcorrelation = ten
361  rbesttotalcorrelation = ten ! The best mean of all correlations
362  ! The thickness with the lowest figure of merit for each image
363  ibestimagethicknessindex = 1
364 
365  !\/----------------------------------------------------------------------
366  DO jnd = 1,ithicknesscount
367  rtotalcorrelation = zero ! The sum of all individual correlations, initialise at 0
368  DO ind = 1,inooflacbedpatterns
369  rsimulatedimage = rimagesimi(:,:,ind,jnd)
370  rexperimentalimage = rimageexpi(:,:,ind)
371  IF (icorrelationflag.EQ.3) THEN ! masked correltion, update mask
372  rmaskimage=rimagemask(:,:,ind)
373  END IF
374 
375  ! image processing
376  SELECT CASE (iimageprocessingflag)
377  !CASE(0) !no processing
378  CASE(1)! square root before perfoming correlation
379  rsimulatedimage=sqrt(rsimulatedimage)
380  rexperimentalimage=sqrt(rimageexpi(:,:,ind))
381  CASE(2)! log before performing correlation
382  WHERE (rsimulatedimage.GT.tiny)
383  rsimulatedimage=log(rsimulatedimage)
384  ELSEWHERE
385  rsimulatedimage = tiny
386  END WHERE
387  WHERE (rexperimentalimage.GT.tiny)
388  rexperimentalimage = log(rimageexpi(:,:,ind))
389  ELSEWHERE
390  rexperimentalimage = tiny
391  END WHERE
392 
393  END SELECT
394 
395  ! Correlation type
396  SELECT CASE (icorrelationflag)
397  CASE(0) ! Phase Correlation
398  rimagecorrelation=one-& ! NB Perfect Correlation = 0 not 1
399  phasecorrelate(rsimulatedimage,rexperimentalimage,&
400  ierr,2*ipixelcount,2*ipixelcount)
401  CASE(1) ! Residual Sum of Squares (Non functional)
402  rimagecorrelation = residualsumofsquares(&
403  rsimulatedimage,rimageexpi(:,:,ind),ierr)
404  CASE(2) ! Normalised Cross Correlation
405  rimagecorrelation = one-& ! NB Perfect Correlation = 0 not 1
406  normalised2dcrosscorrelation(rsimulatedimage,rexperimentalimage,ierr)
407  CASE(3) ! Masked Cross Correlation
408  IF (iter.LE.0) THEN
409  ! we are in baseline sim or simplex initialisation: do a normalised2D CC
410  rimagecorrelation = one-&
411  normalised2dcrosscorrelation(rsimulatedimage,rexperimentalimage,ierr)
412  ELSE ! we are refining: do a masked CC
413  rimagecorrelation = one-& ! NB Perfect Correlation = 0 not 1
414  maskedcorrelation(rsimulatedimage,rexperimentalimage,rmaskimage,ierr)
415  END IF
416  END SELECT
417 
418  CALL message(lxl,dbg6,"For Pattern ",ind,", thickness ",jnd)
419  CALL message(lxl,dbg6," the FoM = ",rimagecorrelation)
420 
421  ! Determines which thickness matches best for each LACBED pattern
422  ! which is later used to find the range of viable thicknesses
423  IF(rimagecorrelation.LT.rbestcorrelation(ind)) THEN
424  rbestcorrelation(ind) = rimagecorrelation
425  ibestimagethicknessindex(ind) = jnd
426  END IF
427  rtotalcorrelation = rtotalcorrelation + rimagecorrelation
428  END DO
429  rtotalcorrelation=rtotalcorrelation/REAL(inooflacbedpatterns,rkind)
430 
431  ! Determines which thickness matches best
432  IF(rtotalcorrelation.LT.rbesttotalcorrelation) THEN
433  rbesttotalcorrelation = rtotalcorrelation
434  ibestthicknessindex = jnd
435  END IF
436 
437  CALL message(lm,dbg6,"Specimen thickness number ",ibestthicknessindex)
438  CALL message(lm,dbg6,"Figure of merit ",rtotalcorrelation)
439 
440  END DO
441  !/\----------------------------------------------------------------------
442 
443  ! The figure of merit, global variable
444  rfigureofmerit = rbesttotalcorrelation
445 
446  !?? Alternative method below, RWeightingCoefficients not used and has been removed
447  !?? assume that the best thickness is given by the mean of individual thicknesses
448  !IBestThicknessIndex = SUM(IBestImageThicknessIndex)/INoOfLacbedPatterns
449  !RBestThickness = RInitialThickness + (IBestThicknessIndex-1)*RDeltaThickness
450  !RFigureofMerit = SUM(RBestCorrelation*RWeightingCoefficients)/&
451  ! REAL(INoOfLacbedPatterns,RKIND)
452 
453  rbestthickness = rinitialthickness +(ibestthicknessindex-1)*rdeltathickness
454  rthicknessrange=( maxval(ibestimagethicknessindex)-&
455  minval(ibestimagethicknessindex) )*rdeltathickness
456  ! Output to screen, duplicate of felixrefine output
457  WRITE(sprintstring,fmt='(A16,F7.2,A1)') "Figure of merit ",100*rbesttotalcorrelation,"%"
458  CALL message(ls,sprintstring)
459  WRITE(sprintstring,fmt='(A19,I4,A10)') "Specimen thickness ",nint(rbestthickness)," Angstroms"
460  CALL message(ls,sprintstring)
461  WRITE(sprintstring,fmt='(A16,I4,A10)') "Thickness range ",nint(rthicknessrange)," Angstroms"
462  CALL message(ls,sprintstring)
463 
464  RETURN
465 
466  END SUBROUTINE figureofmeritandthickness

533  ralpha = rindependentvariable(ind)
534  CASE(2)
535  rbeta = rindependentvariable(ind)
536  CASE(3)
537  rgamma = rindependentvariable(ind)
538  END SELECT
539  CASE(8)
540  rconvergenceangle = rindependentvariable(ind)
541  CASE(9)
542  rabsorptionpercentage = rindependentvariable(ind)
543  CASE(10)

546  END DO

676  rmin=minval(rimagetoblur)
677  rmax=maxval(rimagetoblur)
678 
679  ! set up a 1D kernel of appropriate size
680  ikernelradius=nint(3*rblurringradius)
681  ALLOCATE(rgauss1d(2*ikernelradius+1),stat=ierr)!ffs
682  rsum=0
683  DO ind=-ikernelradius,ikernelradius
684  rind=REAL(ind)
685  rgauss1d(ind+ikernelradius+1)=exp(-(rind**2)/(2*(rblurringradius**2)))
686  rsum=rsum+rgauss1d(ind+ikernelradius+1)
687  IF(ind==0) ierr=78
688  END DO
689  rgauss1d=rgauss1d/rsum!normalise
690  rtempimage=rimagetoblur*0_rkind !reset the temp image
691 
692  ! apply the kernel in direction 1
693  DO ind = -ikernelradius,ikernelradius
694  IF (ind.LT.0) THEN
695  rshiftimage(1:2*ipixelscount+ind,:)=rimagetoblur(1-ind:2*ipixelscount,:)
696  DO jnd = 1,1-ind!edge fill on right
697  rshiftimage(2*ipixelscount-jnd+1,:)=rimagetoblur(2*ipixelscount,:)
698  END DO
699  ELSE
700  rshiftimage(1+ind:2*ipixelscount,:)=rimagetoblur(1:2*ipixelscount-ind,:)
701  DO jnd = 1,1+ind!edge fill on left
702  rshiftimage(jnd,:)=rimagetoblur(1,:)
703  END DO
704  END IF
705  rtempimage=rtempimage+rshiftimage*rgauss1d(ind+ikernelradius+1)
706  END DO
707 
708  ! make the 1D blurred image the input for the next direction
709  rimagetoblur=rtempimage
710  rtempimage=rimagetoblur*0_rkind ! reset the temp image
711 
712  ! apply the kernel in direction 2
713  DO ind = -ikernelradius,ikernelradius
714  IF (ind.LT.0) THEN
715  rshiftimage(:,1:2*ipixelscount+ind)=rimagetoblur(:,1-ind:2*ipixelscount)
716  DO jnd = 1,1-ind!edge fill on bottom
717  rshiftimage(:,2*ipixelscount-jnd+1)=rimagetoblur(:,2*ipixelscount)
718  END DO
719  ELSE
720  rshiftimage(:,1+ind:2*ipixelscount)=rimagetoblur(:,1:2*ipixelscount-ind)
721  DO jnd = 1,1+ind!edge fill on top
722  rshiftimage(:,jnd)=rimagetoblur(:,1)
723  END DO
724  END IF

726  END DO

730  rtempimage=rtempimage-minval(rtempimage)
