73  ! Crystal Lattice Vectors: orthogonal reference frame in Angstrom units

85  -cos(rgamma)*cos(rgamma)+two*cos(ralpha)*cos(rbeta)*cos(rgamma)) / sin(rgamma))

91  two*cos(ralpha)*cos(rbeta)*cos(rgamma))
92  END IF

97  dot_product(rbveco/dot_product(rbveco,rbveco),rcveco/dot_product(rcveco,rcveco))*&
98  dot_product(rcveco/dot_product(rcveco,rcveco),raveco/dot_product(raveco,raveco))

112  ! Set up Reciprocal Lattice Vectors: orthogonal reference frame in 1/Angstrom units
113  ! RarDirO,RbrDirO,RcrDirO vectors are reciprocal lattice vectors
114  ! 2pi/a, 2pi/b, 2pi/c in an orthogonal frame
115  ! Note that reciprocal lattice vectors have two pi included,
116  ! we are using the physics convention exp(i*g.r)
117  rarveco= twopi*cross(rbveco,rcveco)/dot_product(rbveco,cross(rcveco,raveco))
118  rbrveco= twopi*cross(rcveco,raveco)/dot_product(rcveco,cross(raveco,rbveco))

129  rcrveco(ind) = zero
130  END IF
131  ENDDO

134  ! to orthogonal reference frame (Angstrom units)
135  rtmatc2o(:,1)= raveco(:)
136  rtmatc2o(:,2)= rbveco(:)
137  rtmatc2o(:,3)= rcveco(:)
138 
139  ! RXDirC,RYDirC,RZDirC vectors are the reciprocal lattice vectors
140  ! that define the x-axis of the
141  ! diffraction pattern and the beam direction, they are given in felix.inp
142  ! RXDirO,RYDirO,RZDirO vectors are UNIT reciprocal lattice vectors parallel
143  ! to the above in an orthogonal frame
144  rxdiro= rxdirc(1)*rarveco + rxdirc(2)*rbrveco + rxdirc(3)*rcrveco

147  rzdiro= rzdiro/sqrt(dot_product(rzdiro,rzdiro))
148  rydiro= cross(rzdiro,rxdiro)
149 
150  ! RTmatO2M transforms from orthogonal to microscope reference frame
151  rtmato2m(1,:)= rxdiro(:)
152  rtmato2m(2,:)= rydiro(:)
153  rtmato2m(3,:)= rzdiro(:)
154 
155  ! Unit normal to the specimen in REAL space
156  ! This is used in diffraction pattern calculation SUBROUTINE
157  rnormdirm = matmul(rtmato2m,matmul(rtmatc2o,rnormdirc))
158  rnormdirm = rnormdirm/sqrt(dot_product(rnormdirm,rnormdirm))
159 
160  ! now transform from crystal reference frame to orthogonal and then to microscope frame
161 
162  ! RaVecM, RbVecM, RbVecM unit cell vectors in Angstrom units in the microscope frame
163  ravecm= matmul(rtmato2m,raveco)
164  rbvecm= matmul(rtmato2m,rbveco)
165  rcvecm= matmul(rtmato2m,rcveco)
166

168  ! Note that reciprocal lattice vectors do NOT have two pi included,

170  rarvecm= twopi*cross(rbvecm,rcvecm)/dot_product(rbvecm,cross(rcvecm,ravecm))
171  rbrvecm= twopi*cross(rcvecm,ravecm)/dot_product(rcvecm,cross(ravecm,rbvecm))
172  rcrvecm= twopi*cross(ravecm,rbvecm)/dot_product(ravecm,cross(rbvecm,rcvecm))
