68  SUBROUTINE readdm3tagsandimage( SFilePath, IYPixels, IXPixels, IErr, RImageMatrixDM3 )
69 
70  ! NB The file is read-in with big endian to match the image data format, however
71  ! the tag data is little endian format. The tag characters are read-in byte by byte and
72  ! the pixel size is read-in incorrectly and then converted to the little endian.
73 
74  !?? despite the pixel sizes RImageMatrixDM3 is outputted as square using max side, refer to module notes above
75 
76  CHARACTER(*),INTENT(IN) :: SFilePath
77  INTEGER(4), INTENT(IN) :: IYPixels, IXPixels !?? despite these, RImageArray is outputted as square
78  INTEGER(4),INTENT(OUT) :: IErr
79  REAL(4),INTENT(OUT) :: RImageMatrixDM3(max(iypixels,ixpixels),max(iypixels,ixpixels))
80  ! RImageMatrixDM3( x_pixel, y_pixel )
81 
82  ! parameters (some of these could be inputs but currently unncessary)
83  LOGICAL,PARAMETER :: &
84  LGetImageData = .true.,&
85  lprinttags = .false.,&
86  lprintingallowed = .false. ! if false, excluding errors, printing to terminal is suppressed
87  INTEGER(1),PARAMETER :: &
88  ICorrespondingImageDataTag = 2 ! specifies which 'Data' tag corresponds to image data
89 
90  ! local variables
91  INTEGER(4) :: i,j, INoOfBytes, INoOfTags, INoOfDataTags, INoOfPreDataBytes, Iy, Ix
92  LOGICAL :: LFindingTags, LReadingImageData, LLookingForDataTags
93  CHARACTER(36) :: STagLabel
94  INTEGER(1) :: IPrevious4Bytes(4), IPreviousBytes(40), IByte
95  INTEGER(4) :: I4bytePreData, IDataLengthBigEndian
96  REAL(4) :: RDataBytes ! image data is assumed to be 32bit float type Little Endian
97  ! intialise output variables
98  rimagematrixdm3=0
99  ierr=0
100  ! intialise local variables
101  lfindingtags=.true.; lreadingimagedata=.false.; llookingfordatatags=.true.
102  inoofbytes=0; inooftags=0; inoofpredatabytes=0; inoofdatatags=0; iy=0; ix=1;
103  iprevious4bytes=0; ipreviousbytes=0; i4bytepredata=0;
104  rdatabytes=0; idatalengthbigendian=0;
105 
106  OPEN(unit=1,file=sfilepath,status='OLD',access='STREAM',action='READ',iostat=ierr, convert='LITTLE_ENDIAN')
107  IF(ierr.NE.0) THEN
108  WRITE(*,'(A)') 'Error in ReadDM3TagsAndImage(). Opening .dm3 file, file path =', trim(sfilepath)
109  RETURN
110  ENDIF
111 
112  DO i = 1,4000000 ! should exit due to end of file before this is reached
113  ! each iterationa a set of bytes are read in from the file
114  ! depending upon what is being read-in, this can be 1-4 bytes at a time
115 
116  IF( lfindingtags ) THEN ! read another byte to search and find tags
117 
118  ! remove oldest (leftmost) byte and move list of bytes left by 1
119  ipreviousbytes = cshift(ipreviousbytes,1)
120  iprevious4bytes = cshift(iprevious4bytes,1)
121 
122  ! read byte from file
123  READ(1,iostat=ierr) ipreviousbytes(SIZE(ipreviousbytes)) ! read byte into end of IPreviousBytes array
124  IF(ierr.LT.0) THEN ! End of file reached
125  IF(lprinttags.AND.lprintingallowed) WRITE(*,'(A43,I15)') 'End Of File Reached, Total bytes = ',inoofbytes
126  ierr=0
127  EXIT
128  ENDIF
129  inoofbytes = inoofbytes + 1
130  iprevious4bytes(4) = ipreviousbytes(SIZE(ipreviousbytes))
131 
132  ! check if at a DM3 tag delimiter '%%%%' ('%' is 37 in ASCII)
133  ! and read tag label
134  IF( all(iprevious4bytes.EQ.[37,37,37,37]) ) THEN
135  inooftags = inooftags + 1
136  ! convert bytes before delimiter into corresponding tag label
137  staglabel = ''
138  DO j = SIZE(ipreviousbytes)-4,1,-1
139  ! only convert ASCII printable bytes into characters for labels
140  IF( 32.LE.ipreviousbytes(j) .AND. ipreviousbytes(j).LE.126 ) THEN
141  staglabel = char(ipreviousbytes(j))//trim(staglabel)
142  ELSE
143  EXIT ! non ASCII character found so no longer reading label
144  ENDIF
145  ENDDO
146  IF(lprinttags.AND.lprintingallowed) WRITE(*,'(A36,2x,I5,I15)') staglabel,inooftags,inoofbytes
147  IF(lgetimagedata.AND.llookingfordatatags) THEN
148  IF(staglabel.EQ.'Data') inoofdatatags = inoofdatatags + 1
149  IF(inoofdatatags.EQ.icorrespondingimagedatatag) THEN
150  llookingfordatatags=.false.
151  IF(lprintingallowed) WRITE(*,'(A)') 'Now reading image data'
152  lfindingtags = .false.
153  lreadingimagedata = .true.
154  ENDIF
155  ENDIF
156  ENDIF
157 
158  ELSEIF ( lreadingimagedata ) THEN ! read another set of bytes of image data
159 
160  ! There are 16 pre-data bytes including an integer of the data array length
161  ! which should match the pixel size
162  IF(inoofpredatabytes.LT.16) THEN
163  READ(1,iostat=ierr) i4bytepredata
164  IF(ierr.NE.0) THEN ! error reading dm3
165  WRITE(*,'(A,I0)') 'Error in ReadDM3TagsAndImage(). Reading pre data bytes, Byte number = ',inoofbytes
166  RETURN
167  ENDIF
168  IF(inoofpredatabytes.EQ.12) THEN
169  ! Convert data length integer to big endian format
170  CALL mvbits( i4bytepredata, 24, 8, idatalengthbigendian, 0 )
171  CALL mvbits( i4bytepredata, 16, 8, idatalengthbigendian, 8 )
172  CALL mvbits( i4bytepredata, 8, 8, idatalengthbigendian, 16 )
173  CALL mvbits( i4bytepredata, 0, 8, idatalengthbigendian, 24 )
174  IF(idatalengthbigendian.NE.iypixels*ixpixels) THEN ! error
175  ierr=1
176  WRITE(*,'(A,I0)') 'Error in ReadDM3TagsAndImage(). Data array length does not match inputted pixel size.'
177  WRITE(*,'(A,I0,A,I0)') 'Error in ReadDM3TagsAndImage(). Data array length = ',idatalengthbigendian,', IYPixels*IXPixels = ', iypixels*ixpixels
178  RETURN
179  ENDIF
180  ENDIF
181  inoofbytes = inoofbytes + 4
182  inoofpredatabytes = inoofpredatabytes + 4
183  ELSE ! read image data bytes
184  READ(1,iostat=ierr) rdatabytes
185  IF(ierr.NE.0) THEN ! error reading dm3
186  WRITE(*,'(A,I0)') 'Error in ReadDM3TagsAndImage(). while reading data, Byte number = ',inoofbytes
187  RETURN
188  ENDIF
189  inoofbytes = inoofbytes + 4
190  iy = iy + 1
191  IF(iy .EQ. iypixels + 1) THEN
192  iy = 1
193  ix = ix + 1
194  ENDIF
195  IF(ix .LT. ixpixels + 1) THEN
196  rimagematrixdm3(ix,iy) = rdatabytes
197  ELSE ! reached end of image data
198  lfindingtags = .true.
199  lreadingimagedata = .false.
200  ENDIF
201  ENDIF
202 
203  ELSE ! error, this should never happen
204  ierr=1; WRITE(*,'(A)') 'Error in ReadDM3TagsAndImage(). Reading DM3, cannot search for tags nor read image data'; RETURN
205 
206  ENDIF
207 
208  ENDDO
209  CLOSE(1)
210 
211  END SUBROUTINE readdm3tagsandimage
