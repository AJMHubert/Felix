251  ! extrapolates by a factor fac through the face of the simplex across from the high point,
252  ! it tries it, and replaces the high point if the new point is better
253  fac1=(1.0-fac)/ndim
254  fac2=fac1-fac
255  DO j=1,ndim
256  ptry(j)=psum(j)*fac1-rsimplexvariable(ihi,j)*fac2
257  ENDDO
258  rsendpacket = [10000.0_rkind, ptry, REAL(iter,rkind)]
259  CALL mpi_bcast(rsendpacket,ndim+2,mpi_double_precision,0,mpi_comm_world,ierr)
260  !-------------------------------------------------------------------

264  IF(l_alert(ierr,"SimplexExtrapolate","SimulateAndFit")) RETURN
265  rytry=rfigureofmerit
266 
267  IF (rytry.LT.y(ihi)) THEN
268  y(ihi)=rytry
269  DO j=1,ndim
270  psum(j)=psum(j)-rsimplexvariable(ihi,j)+ptry(j)
271  rsimplexvariable(ihi,j)=ptry(j)
