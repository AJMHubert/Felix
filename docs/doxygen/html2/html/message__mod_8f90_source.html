214  ! initially set indentation and initial characters to be blank

222  no_tag%LState=.false.; dbg7%LState = .false.; dbg3%LState = .false.;

231  SUBROUTINE setmessagemode ( IWriteFLAG, IErr )
232 
233  INTEGER(IKIND), INTENT(out):: IErr
234  INTEGER(IKIND), INTENT(IN) :: IWriteFLAG
235  INTEGER(IKIND) :: IPrio ! short for priority, acts as IWriteFLAG
236  iprio = iwriteflag
237 
238  !--------------------------------------------------------------------
239  ! Use IWriteFLAG tens component to turn-on a MsgTag
240  !--------------------------------------------------------------------
241 
242  SELECT CASE (iprio)
243  CASE (0:9)
244  no_tag%LState=.false.
245  CASE (10:19)
246  mode1%LState = .true.
247  iprio = iprio - 10
248  sspaces = " "
249  ls%SInitialMsg = " "; lm%SInitialMsg = " ";
250  ll%SInitialMsg = " "; lxl%SInitialMsg = " "
251  CASE (70:79)
252  dbg7%LState = .true.
253  iprio = iprio - 70
254  CASE (30:39)
255  dbg3%LState = .true.
256  iprio = iprio - 30
257  CASE (60:69)
258  dbg6%LState = .true.
259  iprio = iprio - 60
260  CASE (90:99)
261  dbg14%LState = .true.
262  iprio = iprio - 90
263  CASE DEFAULT ! error (minor) - tens component not recognised
264  ierr=1
265  IF(l_alert(ierr,"set_message_mod_mode",&
266  "IWriteFLAG tens component not recognised check felix.inp")) RETURN
267  END SELECT
268  ! After the above, IPrio is now just a single digit INTEGER
269 
270  ! switch lower priorities on depending upon IWriteFLAG digit compoent
271  IF (iprio >= 2) lm%LState = .true.;
272  IF (iprio >= 4) ll%LState = .true.;
273  IF (iprio >= 8) lxl%LState = .true.;
274 
275  END SUBROUTINE
