354  SUBROUTINE readhklfile(IErr)
355 
356  USE mynumbers
357  USE message_mod
358 
359  ! global outputs

372  OPEN(unit = ichinp,file="felix.hkl",status='OLD',err=10)
373  iline = 0
374  ihklselectflag=1
375 
376  ! count the number of lines in felix.hkl:
377  ! this is the number of reflections to output, INoOfLacbedPatterns
378  DO
379  READ(unit= ichinp, end=100, fmt='(a)') dummy1
380  iline = iline+1
381  ENDDO
382 100 inooflacbedpatterns = iline
383  CALL message ( lxl, dbg7, "Number of experimental images to load = ", inooflacbedpatterns)
384 
385  ALLOCATE(rinputhkls(inooflacbedpatterns,ithree),stat=ierr)
386  IF(l_alert(ierr,"ReadHklFile","allocate RInputHKLs")) RETURN
387  ALLOCATE(ioutputreflections(inooflacbedpatterns),stat=ierr) !?? should we allocate here JR
388  IF(l_alert(ierr,"ReadHklFile","allocate IOutputReflections")) RETURN
389 
390  ! read in the hkls
391  rewind(unit=ichinp) ! goes to beggining of felix.hkl file
392  iline = 0
393  DO ind = 1,inooflacbedpatterns
394  ! READ HKL in as String
395  READ(unit= ichinp,fmt='(A)' ) dummy1
396  ! Scan String for h
397  ipos1 = scan(dummy1,'[')
398  ipos2 = scan(dummy1,',')
399  dummy2 = dummy1((ipos1+1):(ipos2-1))
400  READ(dummy2,'(I20)') h
401  ! Scan String for k
402  ipos1 = scan(dummy1((ipos2+1):),',') + ipos2
403  dummy2 = dummy1((ipos2+1):(ipos1-1))
404  READ(dummy2,'(I20)') k
405  ! Scan String for l
406  ipos2 = scan(dummy1((ipos1+1):),']') + ipos1
407  dummy2 = dummy1((ipos1+1):(ipos2-1))
408  READ(dummy2,'(I20)') l
409  ! Convert to REALs
410  iline=iline+1
411  rinputhkls(iline,1) = REAL(h,rkind)
412  rinputhkls(iline,2) = REAL(k,rkind)
413  rinputhkls(iline,3) = REAL(l,RKIND)
414  CALL message ( lxl, dbg7, "RInputHKLs", nint(rinputhkls(iline,:)) )
415  END DO
416 
417  RETURN
418 
419  10 CONTINUE
420  ihklselectflag=0
421  CALL message( ll, "felix.hkl not found, continuing")
422  RETURN
