158  IF(l_alert(ierr,"WriteIterationOutput","CLOSE() output .bin file")) RETURN
159  END DO
160 
161  ! writes out structure.cif

163  IF(l_alert(ierr,"WriteIterationOutput","WriteIterationCIF")) RETURN
164  ierr=0

166  WRITE(filename,*) "StructureFactors.txt"
167  WRITE(fullpath,*) trim(adjustl(path)),'/',trim(adjustl(filename))
168  OPEN(unit=ichout,status='UNKNOWN',file=trim(adjustl(fullpath)))
169 
170  DO ind = 1,nreflections
171  WRITE(ichout,fmt='(3I5.1,2F13.9)') nint(rhkl(ind,:)),cugmat(ind,1)
172  END DO

175

266  SUBROUTINE writeoutvariables(Iter,IErr)
267 
268  USE mynumbers
269  USE message_mod
270 
271  ! global inputs
272  USE ipara, ONLY : iabsorbflag, irefinemode, inoofugs, iugoffset, &
273  irefinementvariabletypes
274  USE rpara, ONLY : rbasisatomposition, rbasisoccupancy, rbasisisodw, &
275  ranisotropicdebyewallerfactortensor, rfigureofmerit, &
276  rabsorptionpercentage, rlengthx, rlengthy, rlengthz, ralpha, rbeta, &
277  rgamma, rconvergenceangle, racceleratingvoltage, rrsosscalingfactor
278  USE cpara, ONLY : cuniqueug
279  USE ichannels, ONLY : ichoutsimplex
280 
281  IMPLICIT NONE
282 
283  INTEGER(IKIND),INTENT(IN) :: Iter
284  INTEGER(IKIND) :: IErr,ind,IStart,IEnd,jnd,ITotalOutputVariables
285  CHARACTER(200) :: SFormat, STotalOutputVariables
286  INTEGER(IKIND),DIMENSION(IRefinementVariableTypes) :: IOutputVariables
287  REAL(RKIND),DIMENSION(:),ALLOCATABLE :: RDataOut
288 
289  ! Need to Determine total no. of variables to be written out
290  ! this is different from the no. of refinement variables
291 
292  IF (iabsorbflag.EQ.2) THEN ! Structure Factors are complex
293  ! so require two output variables each
294  ioutputvariables(1) = irefinemode(1)*2*inoofugs+1
295  ! plus one for proportional absorption
296  ELSE
297  ioutputvariables(1) = irefinemode(1)*2*inoofugs
298  END IF
299  ! Atom Coordinates
300  ioutputvariables(2) = irefinemode(2)*SIZE(rbasisatomposition,dim=1)* &
301  SIZE(rbasisatomposition,dim=2)
302  ! Occupancies
303  ioutputvariables(3) = irefinemode(3)*SIZE(rbasisoccupancy,dim=1)
304  ! Isotropic Debye Waller Factors
305  ioutputvariables(4) = irefinemode(4)*SIZE(rbasisisodw,dim=1)
306  ! Anisotropic Debye Waller Factors
307  ioutputvariables(5) = irefinemode(5)*SIZE(ranisotropicdebyewallerfactortensor)
308  ioutputvariables(6) = irefinemode(6) * 3 ! Lattice Parameters (a,b,c)
309  ioutputvariables(7) = irefinemode(7) * 3 ! Lattice Angles (alpha,beta,gamma)
310  ioutputvariables(8) = irefinemode(8) ! Convergence angle
311  ioutputvariables(9) = irefinemode(9) ! Absorption
312  ioutputvariables(10) = irefinemode(10) ! Accelerating Voltage
313  itotaloutputvariables = sum(ioutputvariables) ! Total Output
314 
315  ALLOCATE(rdataout(itotaloutputvariables),stat=ierr)
316  DO jnd = 1,irefinementvariabletypes
317  IF(irefinemode(jnd).EQ.0) THEN
318  cycle ! The refinement variable type is not being refined, skip
319  END IF
320  IF(jnd.EQ.1) THEN ! It's an atom coordinate refinement
321  istart = 1
322  ELSE
323  istart = sum(ioutputvariables(1:(jnd-1)))+1
324  !?? RB there is probably a better way of doing this
325  END IF
326  iend = sum(ioutputvariables(1:jnd))
327 
328  SELECT CASE(jnd)
329  CASE(1)
330  DO ind = 1,inoofugs
331  istart = (ind*2)-1
332  iend = ind*2
333  rdataout(istart:iend) = [REAL(CUniqueUg(ind+IUgOffset)), &
334  REAL(AIMAG(CUniqueUg(ind+IUgOffset)),RKIND)]
335  END DO
336  rdataout(iend+1) = rabsorptionpercentage!RB last variable is absorption
337  CASE(2)
338  rdataout(istart:iend) = &
339  reshape(transpose(rbasisatomposition),shape(rdataout(istart:iend)))
340  CASE(3)
341  rdataout(istart:iend) = rbasisoccupancy
342  CASE(4)
343  rdataout(istart:iend) = rbasisisodw
344  CASE(5)
345  rdataout(istart:iend) = &
346  reshape(ranisotropicdebyewallerfactortensor,shape(rdataout(istart:iend)))
347  CASE(6)
348  rdataout(istart:iend) = [rlengthx, rlengthy, rlengthz]
349  CASE(7)
350  rdataout(istart:iend) = [ralpha, rbeta, rgamma]
351  CASE(8)
352  rdataout(istart:iend) = rconvergenceangle
353  CASE(9)
354  rdataout(istart:iend) = rabsorptionpercentage
355  CASE(10)
356  rdataout(istart:iend) = racceleratingvoltage
357  CASE(11)
358  rdataout(istart:iend) = rrsosscalingfactor
359  CASE(12)
360  DO ind = 1,inoofugs
361  istart = (ind*2)-1
362  iend = ind*2
363  rdataout(istart:iend) = &
364  [REAL(CUniqueUg(ind+IUgOffset)), REAL(AIMAG(CUniqueUg(ind+IUgOffset)),RKIND)]
365  END DO
366  IF (iabsorbflag.EQ.1) THEN
367  rdataout(iend+1) = rabsorptionpercentage
368  ! RB last variable is proportional absorption
369  END IF
370  END SELECT
371  END DO
372 
373  WRITE(stotaloutputvariables,*) itotaloutputvariables
374  WRITE(sformat,*) "(I5.1,1X,F13.9,1X,"//trim(adjustl(stotaloutputvariables))//"(F13.9,1X))"
375 
376  OPEN(unit=ichoutsimplex,file='iteration_log.txt',form='formatted',status='unknown',&
377  position='append')
378  WRITE(unit=ichoutsimplex,fmt=sformat) iter-1,rfigureofmerit,rdataout
379  CLOSE(ichoutsimplex)
380 
381  END SUBROUTINE writeoutvariables
